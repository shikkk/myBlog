## 1. 实现sizeOf函数，传入一个参数的obj，计算这个obj占用了多少字节
以下为64位存储
* number 占8个字节
* string 每个长度2个字节
* boolean 4个字节
* null 不占内存

```javascript
const testData = {
  a: 111,
  b: 'cccc',
  2222: false,
}

const seen = new WeakSet() // 存对象里重复的参数

function sizeOfObj(obj) {
  if (obj === null) return 0
  let bytes = 0
  // 对象里的key也是占用内存空间的
  const objKey = Object.keys(obj)
  for (let i = 0; i<objKey.length; i++) {
    let key = objKey[i]
    if (typeof obj[key] === 'object' && obj[key] !== null) {
      if (seen.has(obj[key])) {  // 如果重复则不叠加计算
        continue;
      } 
      seen.add(obj[key])
    }

    bytes += calculator(key)
    bytes += calculator(obj[key])
  }
  return bytes
}
function calculator(obj) {
  const objType = typeof obj
  switch (objType) {
    case 'string': {
      return obj.length * 2
    }
    case 'boolean': {
      return 4
    }
    case 'number': {
      return 8
    }
    case 'object': {
      if (Array.isArray(obj)) {
        // 对数组的处理
        // [1, 2, 3]
        // [{x:1},{x:2}]
        return obj.map(calculator).reduce((res, cur) => res + cur, 0)
      } else {
        return sizeOfObj(obj)
      }
    }
    default: {
      return 0
    }
  }
}

console.log(calculator(testData))
```

function isEmptyObject (obj){
    for (let k in obj) {
      if(obj.hasOwnProperty(k)) return false
    }

  };

## 3.实现一个getValue函数，传入a.b.c ....

```javascript
let obj = {
    a: {
      b: {
        c: {
          d: {
            e: 5
          }
        }
      }
    }
  }
function getValue(obj,path) {
    if (typeof path !== 'string') throw new Error('参数传入错误')
    let pathSplit = path.split('.')
    for(let i = 0; i<pathSplit.length; i++) {
      if (obj[pathSplit[i]] !== undefined) {
        obj = obj[pathSplit[i]]
      } else {
        throw new Error('传入路径有误')
      }
    }
    return obj
}

const v1 = getValue(obj, 'a.b.c');
console.log(v1)  // d: {e:5}
```

  for (let i = 0; i<this.length/2;i++) {
    let temp = this[i]   // 1, 2, 3

## 6.判断俩个对象是否相等
```javascript
let obj1 = {
    name: 'zk',
    sex : 27
};
let obj2 = {
  name: 'zk',
  sex : 27
};
function deepEq (obj1,obj2) {
  if (obj1 === obj2) {
    return true
  }else if ((typeof obj1 === 'object' && obj1 !== null) && (typeof obj2 === 'object' && obj2 !== null)) {
    if (Object.keys(obj1).length !== Object.keys(obj2).length) return false
    for (let i in obj1) {
      if (obj1.hasOwnProperty(i)) {
        if (!deepEq(obj1[i],obj2[i])) return false
      } else {
        return false
      }
    }
    return true
  } else {
    return false
  }
}

console.log(deepEq(obj1,obj2))
```
